redis常用命令

# redis 常用命令

# 登录启动切换等...
#配置文件后台启动redis
redis-server /xxx/xxx/redis.conf

#查看redis启动进程 6379
ps -ef|grep redis

#客户端访问redis【当有多个redis时，需要指定端口号】
redis-cli -p 6379

#关闭redis【当有多个redis时，需要指定端口号】
redis-cli -p 6379 shutdown

#关闭redis【当在登录状态时】
shutdown
exit

#切换redis数据库
select 8<dbID>			

#查看当前数据库的key的数量
dbsize

#清空当前数据库
flushdb

#通杀全部库
flushall

# key
#查看当前库的所有key【支持模糊匹配】
keys *		[keys *1]

#判断某个key k1 是否存在
exits <k1>

#获取某个key k1 的值
get <k1>

#查看某个key k1 属性值的类型
type <k1>

#删除某个key k1，阻塞式删除【等k1删除完了，采取做其他事】
del k1

#删除某个key k1，非阻塞式删除【若有其他事务，可以晾在那儿，等会儿删,异步操作】
unlink k1

#设置过期时间【k2 10s expire】
expire k2 10

#查看key过期时间【返回-2则过期】全程：time to life
ttl k2

# string
#关联键和值【只能一个一个设置】【覆盖】
set <key> <value>

#关联键和值【只能一个一个设置】【不覆盖，原子操作】全称：set if no exists
setnx <key> <value>

#批量设置键和值【设置多个】【覆盖】全称：multi set
mset <key> <value> <key> <value> <key> <value> ...

#批量设置键和值【设置多个】【不覆盖，原子操作，一个失败全失败】全称：multi set if no exists
msetnx <key> <value> <key> <value> <key> <value> ...

#批量获取键的值 全称：multi set
mget <key> <key> <key> ...

#设置键值对的过期时间 全称：set expiration
setex <key> <seconds/ <value>

#获取键对应值的部分/全部
getrange <key> <startIndex> <endIndex>

#在value中指定位置插入值
setrange <key> <startIndex> <endIndex>

#在value中追加值
append <key> <value>

#数值型value +1 全称：increment key
incr key

#数值型value -1 全称：decrement key
decr key

#在指定数值基础上增加or减少
incrby key num
decrby key num 

#得到该字符串的长度 全称：string length key
strlen key

# list
#链表已经初始化ok，直接使用命令插值就行

#从左边开始进入队列【->】全称：left push
lpush key value value value ...

#从右边开始进入队列【<-】全称：right push
rpush key value value value ...

#从队列右边弹出一个值【>-,值在键在，值亡键亡】全称：right pop
rpop key

#从队列左边弹出一个值【>-,值在键在，值亡键亡】全称：left pop
lpop key

#从①号队列【右边】弹出一个值，左边插入到②号队列中
rpoplpush key1 key2
rpoprpush key1 key2
lpoplpush key1 key2
lpoprpush key1 key2

#获取索引下标的值
lindex key[listname] index

#按索引下表获取元素的值 全称：listname range
lrange key startIndex endIndex

#获取链表队列的长度【元素的个数】 全称：listname length
llen key

#在链表队列value值的后面插入一个newvalue值 全称：listname insert
linsert key before value newvalue

# n>0 从左向右删除n个元素。
# n<0 从右向左删除n个元素。
# n=0 删除所有值为指定值的值。
lrem key[listname] n z指定值

#设置替换值【将listname的index处的值替换位value】
lset key index value
